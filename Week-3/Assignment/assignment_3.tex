\begin{align}
	f_1(x) &= sin ^ 2(x ^ 4)\\

\end{align}

\subsection{Verify whether or not each of these functions satisfy Laplace's equation}

\begin{align}
f_1(x) &= 0\\
f_2(x, y) &= \sin(x) + \cos(x)\\
f_3(x, y, z) &= \sin(x^2) - \cos(xy) + \sin(x)\\
f_4(x, y) &= x^2 + y^2 - 2xy\\
f_5(x, y, z) &= 2x + 2z - 2y\\
f_6(x, y) &= e^{-2xy} - 2 e^{2xy}
\end{align}
